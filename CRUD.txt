CREATE OR REPLACE PACKAGE SCOTT.PACK_CATEGORIA

AS

--PROCEDIMIENTO INSERTAR

PROCEDURE INSERTAR(

PR_id_categoria IN CATEGORIA.id_categoria%TYPE,

PRO_nombre_categoria IN CATEGORIA.nombre_categoria%TYPE,

PR_descripcion IN CATEGORIA.descripcion%TYPE,

PR_RESPUESTA  OUT NUMBER,

P_MENSAJE OUT VARCHAR2

);

IS
 
 V_VERIFICADOR NUMBER:= 0;
BEGIN 

-- DUPLICACION DE CATEGORIA 

SELECT COUNT(*) INTO V_VERIFICADOR 
FROM  SCOOT.CATEGORIA
 WHERE id_categoria=PR_id_categoria;

IF(V_VERIFICADOR>0)
THEN
RAISE_APPLICATION_ERROR(-200001,'CODIGO CATEGORIA DUPLICADO');
END IF;

EXCEPTION

WHEN NO_DATA_FOUND THEN
DBMS_OUTPU.PUT_LINE('SIN DEVOLVER');

WHEN OTHERS THEN
DBMS_OUTPUT.PUT_LINE('ERRO');
END;
--PROCEDIMIENTO ACTUALIZAR

PROCEDURE ACTUALIZAR
(
PR_id_categoria IN CATEGORIA.id_categoria%TYPE,

PRO_nombre_categoria IN CATEGORIA.nombre_categoria%TYPE,

PR_descripcion IN CATEGORIA.descripcion%TYPE,

PR_RESPUESTA  OUT NUMBER,

PR_MENSAJE OUT VARCHAR2

);

IS

V_VERIFICADOR NUMER :=0;

BEGIN

--VERIFICANDO PARA LA ACTUALIZACION

SELECT COUNT(*) INTO V_VERIFICADOR

FROM SCOOT.CATEGORIA 

WHERE id_categoria=PR_id_categoria;

IF(V_VERIFICAR = 0)
THEN 
RAISE_APPLICATION_ERRO(-20002,'NO ENOCONTRAD');
END IF;

--ACTUALIZANDO

UPDATE SCOOT.CATEGORIA SET 

nombre_categoria    =  PR_nombre_categoria,
 
descripcion         = PR_descripcion

WHERE id_categoria=PR_id_categoria;


EXCEPTION

WHEN NO_DATA_FOUND THEN
DBMS_OUTPU.PUT_LINE('SIN ACTUALIZAR');

WHEN OTHERS THEN
DBMS_OUTPUT.PUT_LINE('ERROR');
END;


-------ELIMINANDO 

PROCEDURE ELIMINAR(
PR_id_categoria  IN  CATEGORIA.id_categoria%TYPE;

PR_RESPUESTA OUT NUMBER,

PR_MENSAJE OUT VARCHAR2
);

IS

V_VERIFICADOR = 0;

BEGIN

--VALIDANDO CODIGO 

SELECT COUNT(*) INTO V_VERIFICADOR 

FROM SCOTT.CATEGORIA 

WHERE id_categoria=PR_id_categoria;

IF (V_VERIFICADOR=0 )
THEN

RAISE_APPLICATION_ERROR(-20003,'SIN CLIENTE');

END IF;

--ELIMINANDO REGISTRO

DELETE FROM SCOTT.CATEGORIA 
WHERE id_categori=PR_id_categoria;



EXCEPTION

WHEN NO_DATA_FOUND THEN
DBMS_OUTPU.PUT_LINE('SIN ELIMINAR');

WHEN OTHERS THEN
DBMS_OUTPUT.PUT_LINE('ERROR');

END;
